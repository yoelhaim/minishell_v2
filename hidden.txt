void test(t_node **list)
{
	t_node *cmd;
	cmd = *list;
	while (cmd)
	{
		printf("\033[0;31m-----------------------------------\033[0m\n");
		printf("| value => %s          |\n", cmd->val);
		printf("| type => %d           |", cmd->type);
		printf("\n\033[32m-----------------------------------\033[0m\n");
		cmd = cmd->next;
	}
	
		
}
